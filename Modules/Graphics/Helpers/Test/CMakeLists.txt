include(CTest)
include(ParseAndAddCatchTests)

    Main.cpp
    ActionCameraTest.cpp
)

    MethaneGraphicsHelpers
    Catch
)

if (METHANE_RUN_TESTS_DURING_BUILD)

    PROPERTIES
    FOLDER Tests
)

    RUNTIME
        DESTINATION Tests
        COMPONENT Test
)
